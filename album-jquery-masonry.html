用jQuery简单实现照片墙功能

Posted on 2012-04-03 | Edited on 2019-05-07 | In 折腾

对wordpress主题的折腾实在没有什么动力，什么AJAX动态加载之类的，主要感觉主题能看就行。由于之前是使用ImageVue建立相册，对于我来说功能太复杂，只是想实现照片墙的功能和点击放大，觉得自己写也差不多吧。 开始：在网上查找一个页面混合排版的插件，发现了一个jQuery Masonry的插件，心想就用jQuery来实现吧。 基于jQuery Masonry的照片墙是挺简单的，不需要管页面的布局，首先加载一个jQuery1.4以上的版本，用jquery.masonry.min.js来管理页面布局，用slimbox2.js来显示大图。思路就是这样，下面就贴核心的代码吧

jQuery(document).ready(function($){
//动态加载class为box的图片
var $container = $(‘#container’);
$container.imagesLoaded( function(){
$container.masonry({
itemSelector : ‘.box’,
isAnimated: true,
});
});

//实现图片选中突出
$(".box").hover(function(){
    $(this).fadeTo("fast", 0.75);
},function(){
       $(this).fadeTo("fast", 1.0);
});

/\*\*
\* 下面主要实现点击随机显示图片的功能
*/    
//图片的相对路径
var blogString = 'http://farm5.static.flickr.com/4113/5013039951\_3a47ccd509.jpg+http://farm5.static.flickr.com/4131/5013039885\_0d16ac87bc.jpg+http://farm5.static.flickr.com/4086/5013039583\_26717f6e89.jpg+http://farm5.static.flickr.com/4146/5013646070\_f1f44b1939.jpg+http://farm5.static.flickr.com/4144/5013039541\_17f2579e33.jpg+http://farm5.static.flickr.com/4153/5013039741\_d860fb640b.jpg+http://farm5.static.flickr.com/4113/5013039697\_a15e41fcd8.jpg+http://farm5.static.flickr.com/4124/5013646314\_c7eaf84918.jpg+http://farm5.static.flickr.com/4089/5013040075_bac12ff74e.jpg';
var blog = blogString.split("+");
var blogLen = blog.length;

$('#random').click(function(){
    var whichSentence = '';
    var allText = '';
    var contain = '';
    $container.children().each(function(){
        //生成随机数选取图片
        whichSentence = Math.floor( Math.random() * blogLen );
        allText = blog\[whichSentence\];
        //防止图片重复出现
        while(contain.indexOf(allText) >=0)    {
        whichSentence = Math.floor( Math.random() * blogLen );
        allText = blog\[whichSentence\];
        }
        contain = contain + allText;
        //修改图片的地址            
        $(this).children().children().attr("src",allText);
        $(this).children().attr("href",allText);
    });
    //再次等待图片的加载
    $container.imagesLoaded( function(){
      $container.masonry({
        itemSelector : '.box',
        isAnimated: true,
      });
    });
    //重新排版
    $container.prepend($container.children().last().prev().prev());
    $container.prepend($container.children().last().prev().prev().prev());
    $container.prepend($container.children().last());
    $container.prepend($container.children().last().prev());
    $container.masonry( 'reload' );
});

});
